    <script>
        // استبدل هذا الرابط بالرابط اللي خدته من Google Apps Script
        const SCRIPT_URL = 'هنا_حط_رابط_Google_Apps_Script_الخاص_بك';
        
        let userLat = null;
        let userLng = null;
        let userAccuracy = null;
        let locationAttempts = 0;
        const MAX_ATTEMPTS = 3;

        // عناصر الصفحة
        const locationMessage = document.getElementById('locationMessage');
        const locationDetails = document.getElementById('locationDetails');
        const coordsDisplay = document.getElementById('coordsDisplay');
        const accuracyDisplay = document.getElementById('accuracyDisplay');
        const submitBtn = document.getElementById('submitBtn');
        const resultDiv = document.getElementById('result');
        const refreshBtn = document.getElementById('refreshLocationBtn');

        // دالة الحصول على الموقع
        function getLocation() {
            locationMessage.innerHTML = '🔄 جاري الحصول على موقعك...';
            locationMessage.className = 'location-text';
            locationDetails.style.display = 'none';
            submitBtn.disabled = true;
            
            if (!navigator.geolocation) {
                showLocationError('المتصفح لا يدعم خاصية تحديد الموقع');
                return;
            }

            const options = {
                enableHighAccuracy: true,
                timeout: 10000,
                maximumAge: 0
            };

            navigator.geolocation.getCurrentPosition(
                // نجاح
                (position) => {
                    userLat = position.coords.latitude;
                    userLng = position.coords.longitude;
                    userAccuracy = position.coords.accuracy;
                    
                    // عرض الموقع
                    locationMessage.innerHTML = '✅ تم تحديد موقعك بنجاح!';
                    locationMessage.className = 'location-text success';
                    
                    coordsDisplay.innerHTML = `خط العرض: ${userLat.toFixed(6)} | خط الطول: ${userLng.toFixed(6)}`;
                    
                    let accuracyText = '';
                    let accuracyClass = '';
                    if (userAccuracy < 20) {
                        accuracyText = 'دقة عالية جداً';
                        accuracyClass = 'badge-success';
                    } else if (userAccuracy < 50) {
                        accuracyText = 'دقة متوسطة';
                        accuracyClass = 'badge-warning';
                    } else {
                        accuracyText = 'دقة منخفضة';
                        accuracyClass = 'badge-warning';
                    }
                    
                    accuracyDisplay.innerHTML = `دقة الموقع: <span class="badge ${accuracyClass}">${userAccuracy.toFixed(0)} متر (${accuracyText})</span>`;
                    
                    locationDetails.style.display = 'block';
                    submitBtn.disabled = false;
                    
                    // إعادة تعيين المحاولات
                    locationAttempts = 0;
                },
                // خطأ
                (error) => {
                    locationAttempts++;
                    let errorMessage = '';
                    
                    switch(error.code) {
                        case error.PERMISSION_DENIED:
                            errorMessage = '❌ تم رفض الإذن بالوصول للموقع. الرجاء السماح بالوصول للموقع من إعدادات المتصفح';
                            break;
                        case error.POSITION_UNAVAILABLE:
                            errorMessage = '❌ معلومات الموقع غير متوفرة. تأكد من تشغيل GPS';
                            break;
                        case error.TIMEOUT:
                            errorMessage = '❌ انتهى وقت محاولة الحصول على الموقع. حاول مرة أخرى';
                            break;
                        default:
                            errorMessage = '❌ حدث خطأ غير معروف في تحديد الموقع';
                    }
                    
                    showLocationError(errorMessage);
                    
                    // إذا كان العدد أقل من الحد الأقصى، حاول مرة أخرى
                    if (locationAttempts < MAX_ATTEMPTS) {
                        setTimeout(() => {
                            locationMessage.innerHTML = `🔄 محاولة ${locationAttempts + 1} من ${MAX_ATTEMPTS}...`;
                            getLocation();
                        }, 2000);
                    }
                },
                options
            );
        }

        function showLocationError(message) {
            locationMessage.innerHTML = message;
            locationMessage.className = 'location-text error';
            locationDetails.style.display = 'none';
            submitBtn.disabled = true;
        }

        // عند تحميل الصفحة، نحاول الحصول على الموقع
        window.addEventListener('load', getLocation);

        // تحديث الموقع
        refreshBtn.addEventListener('click', getLocation);

        // عند إرسال النموذج
        document.getElementById('attendanceForm').addEventListener('submit', async function(e) {
            e.preventDefault();
            
            const name = document.getElementById('studentName').value.trim();
            const studentId = document.getElementById('studentId').value.trim();
            
            if (!name || !studentId) {
                showResult('الرجاء إدخال جميع البيانات', 'error');
                return;
            }
            
            if (!userLat || !userLng) {
                showResult('لم يتم الحصول على الموقع بعد. الرجاء الانتظار', 'warning');
                return;
            }
            
            // تعطيل الزر أثناء الإرسال
            submitBtn.disabled = true;
            submitBtn.innerHTML = '<span class="loading"></span> جاري تسجيل البيانات...';
            
            try {
                const formData = new URLSearchParams();
                formData.append('name', name);
                formData.append('studentId', studentId);
                formData.append('lat', userLat);
                formData.append('lng', userLng);
                formData.append('accuracy', userAccuracy);
                
                const response = await fetch(SCRIPT_URL, {
                    method: 'POST',
                    mode: 'no-cors',
                    headers: {
                        'Content-Type': 'application/x-www-form-urlencoded',
                    },
                    body: formData
                });
                
                // مع no-cors، مش هنقدر نقرأ الـ response
                showResult(
                    '✅ تم تسجيل حضورك بنجاح! شكراً لك', 
                    'success',
                    `https://www.google.com/maps?q=${userLat},${userLng}`
                );
                
                // مسح الحقول
                document.getElementById('studentName').value = '';
                document.getElementById('studentId').value = '';
                
                // تحديث الموقع
                setTimeout(getLocation, 1000);
                
            } catch (error) {
                console.error('Error:', error);
                showResult('❌ حدث خطأ في الاتصال. حاول مرة أخرى', 'error');
            } finally {
                submitBtn.disabled = false;
                submitBtn.innerHTML = 'تسجيل الحضور';
            }
        });

        function showResult(message, type, mapsLink = null) {
            resultDiv.style.display = 'block';
            resultDiv.className = `result ${type}`;
            
            let html = message;
            if (mapsLink) {
                html += `<br><a href="${mapsLink}" target="_blank" class="map-link">📍 عرض موقعك على خرائط Google</a>`;
            }
            
            resultDiv.innerHTML = html;
            
            // إخفاء النتيجة بعد 5 ثواني
            setTimeout(() => {
                resultDiv.style.display = 'none';
            }, 7000);
        }

        // منع إعادة تحميل الصفحة بالسهم
        window.addEventListener('beforeunload', function() {
            // تنبيه اختياري لو عايز تحذر الطلاب قبل الخروج
        });

        // التحقق من صحة الرقم الأكاديمي (رقم مكون من 8 أرقام مثلاً)
        document.getElementById('studentId').addEventListener('input', function(e) {
            this.value = this.value.replace(/[^0-9]/g, '');
        });
    </script>
